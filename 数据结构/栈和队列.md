### 使用两个队列实现一个栈

#### 描述

```
使用两个队列实现一个栈的 push(), pop(), top(), empty();
```
#### 代码实现

```
class Stack
{
    private:
		queue<int> q1;
		queue<int> q2;
    public:
		void push(int num)
		{
			int temp;
			while (!q1.empty())
			{
				temp = q1.front();
				q1.pop();
				q2.push(temp);
			}
			q1.push(num);
			while (!q2.empty())
			{
				temp = q2.front();
				q2.pop();
				q1.push(temp);
			}
		}

		int pop()
		{
			if (q1.empty())
				return -1;
			int temp = q1.front();
			q1.pop();
			return temp;
		}

		int top()
		{
			int temp = -1;
			if (!q1.empty())
			{
				temp = q1.front();
			}
			return temp;
		}

		bool empty()
		{
			return q1.empty();
		}
};
int main()
{
	Stack stack ;
	stack.push(2);
	stack.push(3);
	cout << stack.pop() << endl;
	stack.push(8);
	cout << stack.pop() << endl;
	cout << stack.top() << endl;
}
```
### 用两个栈实现一个队列

#### 描述
用两个栈实现一个队列的push(), pop(), top(), empty()

#### 代码实现

```
class Queue {
	private:
		stack<int> s1;
		stack<int> s2;
    public:
		void push(int num)
		{
			s1.push(num);
		}

		int  pop()
		{
			int temp;
			while (!s1.empty())
			{
				s2.push(s1.top());
				s1.pop();
			}
			temp = s2.top();
			s2.pop();
			while (!s2.empty())
			{
				s1.push(s2.top());
				s2.pop();
			}
			return temp;
		}

		int top()
		{
			int temp;
			while (!s1.empty())
			{
				s2.push(s1.top());
				s1.pop();
			}
			temp = s2.top();
			while (!s2.empty())
			{
				s1.push(s2.top());
				s2.pop();
			}
			return temp;
		}

		bool empty()
		{
			return s1.empty() && s2.empty();
		}
};
int main()
{
	Queue queue;
	queue.push(1);
	queue.push(2);
	queue.push(3);

	cout << queue.pop() << endl;
	cout << queue.pop() << endl;
}

### 实现逆波兰序的计算

#### 描述
输入一个中缀表达式， 输出表达式的计算结果

```
输入:["0","3","/"]
输出:0
```
#### 代码实现


```
int evalRPN(vector<string>& tokens) 
{
        int result;
        if (tokens.size() == 0)
        {
            return 0;
        }

        stack<int> nums;

        for (int i = 0; i < tokens.size(); i++)
        {
            if (tokens[i] != "+" && tokens[i] != "-" && tokens[i] != "*" && tokens[i] != "/")
            {
                int temp = atoi(&tokens[i][0]);
                nums.push(temp);
            }
            else
            {
                int num2 = nums.top();
                nums.pop();
                int num1 = nums.top();
                nums.pop();
                nums.push(getResult(num1, num2, &tokens[i][0]));
            }
        }

        result = nums.top();

        return result;
    }
    
    int getResult(int num1, int num2, char *operation)
    {
    	int result = 0;
    	switch (*operation)
    	{
    		case '+':
    			result = num1 + num2;
    			break;
    		case '-':
    			result = num1 - num2;
    			break;
    		case '*':
    			result = num1 * num2;
    			break;
    		case '/':
    			result = num1 / num2;
    			break;
    	}
    	return result;
    }
```
### 二叉树的前序遍历

#### 描述
前序：根->左->右； 
非递归前序遍历二叉树
#### 代码实现

```
vector<int> preorderTraversal(TreeNode* root)
{
	vector<int> result;
	stack<TreeNode*> node;
	TreeNode* T = root;

	while (!node.empty() || T)
	{
		while (T)
		{
			result.push_back(T->val);
			node.push(T);
			T = T->left;
		}
		if (!node.empty())
		{
			T = node.top();
			node.pop();
			T = T->right;
		}
	}
	return result;
}
```

### 二叉树的中序遍历

#### 描述
中序：左->根->右。
非递归中序遍历二叉树。在二叉树的前序遍历中，先将当前节点输出，然后将当前节点保存到栈中，最后进入其左子树，中序节点可以在前序遍历的代码基础上，将节点值输出的部分加入到进入当前节点右孩子之前。

#### 代码实现

```
vector<int> inorderTraversal(TreeNode* root)
{
	vector<int> result;
	stack<TreeNode*> node;
	TreeNode* T = root;

	while (!node.empty() || T)
	{
		while (T)
		{
			node.push(T);
			T = T->left;
		}
		if (!node.empty())
		{
			T = node.top();
			node.pop();
			result.push_back(T->val);
			T = T->right;
		}
	}
	return result;
}
```

### 二叉树的后续遍历

#### 描述
- 后续：左->右->根；
- 后续的逆序: 根->右->左
- 前序: 根->左->右
- 所以，只要按前序遍历(调整左右节点遍历顺序)，将遍历结果压人另外一个栈，最后将该栈中的元素依次出栈，即可得到后续遍历序列。


#### 代码实现

```
vector<int> postorderTraversal(TreeNode* root)
{
	stack<TreeNode*> node1;
	stack<TreeNode*> node2;
	vector<int> result;
	TreeNode* T = root;
	while (!node1.empty() || T)
	{
		while (T)
		{
			node2.push(T);
			node1.push(T);
			T = T->left;
		}
		if (!node1.empty())
		{
			T = node1.top();
			node1.pop();
			T = T->right;
		}
	}

	while (!node2.empty())
	{
		result.push_back(node2.top()->val);
		node2.pop();
	}
	return result;
}
```
### 包含最小值的栈

#### 描述
实现一个栈的数据结构，能够在O(1)的时间内获取栈中元素的最小值。

#### 代码实现

```
#include<iostream>
#include<stdio.h>
#include<stack>
#include<string>
#include<vector>
using namespace std;

class Stack {
private:
	stack<int> item;
	stack<int> minItem;
public:
	void push(int data)
	{
		item.push(data);
		if (minItem.empty() || minItem.top() > data)
		{
			minItem.push(data);
		}
		else
		{
			minItem.push(minItem.top());
		}
	}

	void pop()
	{
		if (!item.empty())
		{
			item.pop();
		}
		if (!minItem.empty())
		{
			minItem.pop();
		}
	}

	int minValue()
	{
		int result = -1;
		if (!minItem.empty())
		{
			result = minItem.top();
		}
		return result;
	}
};

int main()
{
	Stack t;
	t.push(2);
	t.push(5);
	t.push(1);
	cout << t.minValue() << endl;
	t.pop();
	cout << t.minValue() << endl;
}
```
### 栈的压入，弹出序列

```
#include<iostream>
#include<stdio.h>
#include<stack>

#include<queue>

#include<vector>
using namespace std;

bool isPopSequence(vector<int>& pushS, vector<int>& popS)
{
	int length = popS.size();
	if (length == 0)
	{
		return false;
	}
	int begin = 0;
	int start = 0;
	stack<int> pushTemp;

	while (begin != length - 1)
	{
		if (pushTemp.empty() || pushTemp.top() != popS[begin])
		{
			pushTemp.push(pushS[start++]);
		}
		
		if (pushTemp.top() != popS[begin])
		{
			return false;
		}

		pushTemp.pop();
		begin++;
	}
	if (pushTemp.empty() || start == length - 1)
	{
		return true;
	}
	return false;
}

int main()
{
	vector<int> pushSequence = { 1,2,3,4,5 };
	vector<int> popSequence = {4,3,5,1,2 };

	cout << isPopSequence(pushSequence, popSequence) << endl;
}

### 队列的最大值

```
#include<iostream>
#include<stdio.h>
#include<stack>
#include<string>
#include<queue>
#include<deque>
#include<map>
#include<vector>
using namespace std;

vector<int> maxValueInWindows(vector<int>& nums, int size)
{
	vector<int> result;
	if (nums.size() == 0 || size <= 0)
	{
		return result;
	}
	deque<int> index;
	for (int i = 0; i < size; i++)
	{
		while (!index.empty() && nums[i] >= nums[index.back()])
		{
			index.pop_back();
		}
		index.push_back(i);
	}
	for (int i = size; i < nums.size(); i++)
	{
		result.push_back(nums[index.front()]);
		while (!index.empty() && nums[i] >= nums[index.back()])
		{
			index.pop_back();
		}
		while (!index.empty() && index.front() <= i - size)
		{
			index.pop_front();
		}
		index.push_back(i);
	}
	result.push_back(nums[index.front()]);

	return result;
}

void show(vector<int> nums)
{
	int len = nums.size();
	for (int i = 0; i < len; i++)
	{
		cout << nums[i] << " ";
	}
	cout << endl;
}
int main()
{
	vector<int> nums = { 2,3,4,2,6,2,5,1 };
	vector<int> result = maxValueInWindows(nums, 3);
	show(result);
}

### 拥有最大值的队列

```
#include<iostream>
#include<stdio.h>
#include<stack>
#include<string>
#include<queue>
#include<deque>
#include<map>
#include<vector>
using namespace std;

class queueWithMax
{
private:
	struct internalData
	{
		int value;
		int index;
	};

	deque<internalData> data;
	deque<internalData> maximums;

	int currentIndex;
public:
	queueWithMax() : currentIndex(0){}

	void push_back(int number)
	{
		while (!maximums.empty() && number >= maximums.back().value)
		{
			maximums.pop_back();
		}

		internalData Data = { number, currentIndex };
		data.push_back(Data);
		maximums.push_back(Data);
		currentIndex++;
	}

	void pop_front()
	{
		if (maximums.empty())
		{
			cout << "queue is empty" << endl;
			return;
		}
		if (maximums.front().index == data.front().index)
		{
			maximums.pop_front();
		}
		data.pop_front();
	}

	int maxValue()
	{
		if (maximums.empty())
		{
			cout << "queue is empty" << endl;
			return -1;
		}
		return maximums.front().value;
	}
};

int main()
{
	queueWithMax myQueue;
	myQueue.push_back(2);
	myQueue.push_back(5);
	myQueue.push_back(8);
	cout << myQueue.maxValue() << endl;
}
