### 替换空格
#### 描述
在网络编程中，如果URL参数中含有特殊字符， 比如空格等，则可导致服务器端无法获得正确参数，我们需要将这些字符替换为服务器可以识别的字符。

```
输入: we are happy
输出: we%20are%20happy
```
#### 思路
1. 遍历， 获得原字符串中空格的数量
2. 计算替换后字符串的长度
3. 用两个指针， 从尾到头进行复制，若遇到空格，插入‘%20’


#### 代码实现

```
void  replaceBlank(char str[], int length)
{
	if (str == nullptr || length == 0)
	{
		return;
	}
	int originalLength = 0;
	int numOfBlank = 0;

	while (*str != '\0')
	{
		originalLength++;
		if (*str == ' ')
		{
			numOfBlank++;
		}
		str++;
	}
	int newLength = originalLength + numOfBlank * 2;
	if (newLength > length)
	{
		return;
	}
	int indexOfOriginal = originalLength - 1;
	int indexOfNew = newLength - 1;

	while (indexOfOriginal >= 0 && indexOfNew > indexOfOriginal)
	{
		if (str[indexOfOriginal] == ' ')
		{
			str[indexOfNew--] = '0';
			str[indexOfNew--] = '2';
			str[indexOfNew--] = '%';
		}
		else
		{
			str[indexOfNew--] = str[indexOfOriginal];
		}
		--indexOfOriginal;
	}
}
```
### 打印从1到最大的n位数

#### 描述

```
输入：n=2
输出：1--99
```
#### 思路
- 前求出最大的n位数，再循环打印(大数问题)；
- 利用字符串模拟加1操作，解决大数问题；
- 利用n位数字的全排列；

#### 代码实现


```
void printNumber(char* number)
{
	bool isBeginning0 = true;
	int nLength = strlen(number);

	for (int i = 0; i < nLength; i++)
	{
		if (isBeginning0 && number[i] != '0')
		{
			isBeginning0 = false;
		}
		if (!isBeginning0)
		{
			cout << number[i];
		}
	}
	cout << endl;
}

void print1ToMax(char* number, int length,int index)
{
	if (index == length - 1)
	{
		printNumber(number);
		return;
	}

	for (int i = 0; i < 10; i++)
	{
		number[index + 1] = i + '0';
		print1ToMax(number, length, index + 1);
	}
}


void print1(int n)
{
	if (n <= 0)
	{
		return;
	}
	char* number = new char[n + 1];
	number[n] = '\0';
	for (int i = 0; i < 10; i++)
	{
		number[0] = i + '0';
		print1ToMax(number, n, 0);
	}
	delete[] number;
}
int main()
{
	print1(2);
}

### 字符串匹配

#### 判断两个字符串是否相同

```
输入："abbc"
输入："abbc"
输出：1
输入："abbd"
输入："abbc"
输出：0
```
#### 代码实现

```
bool match(char* str, char* pattern)
{
	if (*str == '\0' && *pattern == '\0')
	{
		return true;
	}
	else if (*str == '\0' || *pattern == '\0')
	{
		return false;
	}
	if (*str == *pattern)
	{
		return match(str + 1, pattern + 1);
	}
	else
	{
		return false;
	}
	return false;
}

int main()
{
	string s1 = "thi1";
	string s2 = "thi2";

	char* str = &s1[0];
	char* pattern = &s2[0];

	cout << match(str, pattern) << endl;
}

int numberOf1(int n)
{
	int result = 0;
	while (n)
	{
		if (n % 10 == 1)
			result++;
		n = n / 10;
	}
	return result;
}

int numberOf1(char* strN, int length)
{
	if (!strN || *strN < '0' || *strN > '9' || *strN == '\0')
	{
		return 0;
	}
	//首位数字
	int first = *strN - '0';

	if (length == 1 && first == 0)
	{
		return 0;
	}
	if (length == 1 && first > 0)
	{
		return 1;
	}

	//假设 strN是 “21345”
	// numFirstDigit是数字10000 - 19999 的第一位的数目
	int numFirstDigit = 0;
	if (first > 1)
	{
		numFirstDigit = pow(10, length - 1);
	}
	else if (first == 1)
	{
		numFirstDigit = atoi(strN + 1) + 1;
	}
	//numOtherDigits 是1346 - 21345除第一位之外的数位中的数目
	int numOtherDigits = first * (length - 1) * pow(10, length - 2);
	//numRecursive 是1 - 1345中的数目
	int numRecursive = numberOf1(strN + 1, length-1);

	return numFirstDigit + numOtherDigits + numRecursive;
}

### 翻转单词中的字符

#### 代码实现

#include<iostream>
#include<stdio.h>

void Reverse(char* begin, char* end)
{
	if (begin == nullptr || end == nullptr)
	{
		return;
	}
	while (begin < end)
	{
		char temp = *begin;
		*begin = *end;
		*end = temp;
		begin++;
		end--;
	}
}

void ReverseSetence(char* str)
{
	if (str == nullptr)
	{
		return;
	}
	char* begin = str;
	char* end = str;
	while (*end != '\0')
	{
		end++;
	}
	end--;
	Reverse(begin, end);
	begin = str;
	end = str;
	while (*begin != '\0')
	{
		if (*begin == ' ')
		{
			begin++;
			end++;
		}
		else if (*end == ' ' || *end == '\0')
		{
			Reverse(begin, --end);
			begin = ++end;
		}
		else
		{
			end++;
		}
	}
}

int main()
{
	string str = "I am a students";
	char* s = &str[0];
	cout << str << endl;
	ReverseSetence(s);
	cout << str << endl;
}

void leftRotateString(char* str, int k)
{
	int length = strlen(str);
	if (length < 0 || k < 0)
	{
		return;
	}
	char* firstBegin = str;
	char* firstEnd = str + k - 1;
	char* secondBegin = str + k;
	char* secondEnd = str + length - 1;

	Reverse(firstBegin, firstEnd);
	Reverse(secondBegin, secondEnd);
	Reverse(firstBegin, secondEnd);
}
